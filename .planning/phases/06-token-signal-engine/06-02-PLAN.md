```typescript
// src/signals/scorer.ts — full interface contract

export interface TokenSignalInputs {
  tokenMint: string;
  smartWalletHolders: Array<{
    walletAddress: string;
    walletScore: number;       // from wallet_metrics.score_total (0-95)
    isCoordinated: boolean;    // has active bundler flag (cleared=false)
    isCurrentHolder: boolean;  // net position > 0 (buy_amt > sell_amt)
  }>;
  buysLast1h: number;          // count of distinct smart wallet buy swaps in last 1h window
  sellsLast1h: number;         // count of distinct smart wallet sell swaps in last 1h window
  totalSmartBuysLast24h: number; // total smart wallet buys in last 24h (for eligibility check)
}

export interface TokenSignalResult {
  signalScore: number;              // 0-100, integer
  signalTier: 'strong' | 'moderate' | 'weak' | 'inactive';
  smartWalletCount: number;         // count of current holders (isCurrentHolder=true)
  buyVelocity1h: number;           // raw buysLast1h (stored as-is for reference)
  buyVelocityScore: number;         // normalized 0-100 sub-score
  exitPressure: number;             // 0.0-1.0, NOT included in signalScore
  pnlWeightedHolderScore: number;  // weighted average of holder walletScores, 0-100
  coordinationDiscount: number;    // multiplier applied (1.0 = no discount, 0.3 = all coordinated)
  coordinatedWalletCount: number;  // count of coordinated current holders
}
```

Formula weights (locked — from CONTEXT.md):
- `WEIGHT_PNL_HOLDER_QUALITY = 0.40`
- `WEIGHT_BUY_VELOCITY = 0.35`
- `WEIGHT_SMART_WALLET_COUNT = 0.25`

Normalization constants (Claude's discretion):
- Buy velocity: 5 buys/hr from smart wallets → velocity sub-score 100 (i.e., `Math.min(100, buysLast1h * 20)`)
- Wallet count: 10 current holders → count sub-score 100 (i.e., `Math.min(100, holderCount * 10)`)
- PnL holder score: average of holder walletScore values (already 0-95 from scoring engine)

Coordination discount: `multiplier = 1.0 - (coordinationRatio * 0.7)`
- All coordinated (ratio=1.0): multiplier = 0.3
- Half coordinated (ratio=0.5): multiplier = 0.65
- None coordinated (ratio=0.0): multiplier = 1.0
- Suppress entirely if all holders are coordinated (return score=0, tier='inactive')

Tier thresholds (Claude's discretion — per RESEARCH.md):
- strong: score >= 65
- moderate: score >= 35
- weak: score < 35 and score > 0
- inactive: score === 0

Minimum floor: MIN_SMART_WALLETS = 2 (current holders with isCurrentHolder=true)

Exit pressure formula: `exitPressure = sellsLast1h / Math.max(1, buysLast1h + sellsLast1h)` — ranges 0.0-1.0
