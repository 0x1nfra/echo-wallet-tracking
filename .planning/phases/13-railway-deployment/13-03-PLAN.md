Output: HeliusCreditExhaustedError in helius.ts; wiring in providers/index.ts that calls monitorLoop.pause() and starts a probe-retry loop.

The ProviderRouter's tryCall* methods catch ALL errors from providers and mark them as cooldown (60s).
HeliusCreditExhaustedError needs different treatment: pause the monitor loop and start a probe loop.

Strategy: Add a `onCreditExhausted` callback parameter to `createProviderRouter()` that wraps HeliusProvider errors at the index.ts level — intercept BEFORE ProviderRouter swallows it.

The simplest correct approach: override the onAllExhausted behavior is not enough. Instead, modify ProviderRouter to re-throw known error types, OR wrap HeliusProvider in index.ts.

Use the wrapper approach in index.ts:

```typescript
import { HeliusCreditExhaustedError } from '../helius.js';
import { monitorLoop } from '../../commands/wallet.js';  // verify exact import path

// Probe config (per research recommendation)
const CREDIT_EXHAUSTION_BASE_DELAY_MS = 5 * 60 * 1000;   // 5 minutes
const CREDIT_EXHAUSTION_MAX_DELAY_MS = 60 * 60 * 1000;   // 60 minutes

function startCreditExhaustionProbe(heliusFetcher: HeliusFetcher): void {
  let delayMs = CREDIT_EXHAUSTION_BASE_DELAY_MS;

  const probe = async () => {
    console.log(`[provider] Helius credit exhaustion probe in ${delayMs / 60000}m...`);
    await new Promise(resolve => setTimeout(resolve, delayMs));
    try {
      // Probe: single lightweight request — fetchOnePage with limit 1 on a known address
      await heliusFetcher.fetchOnePage('So11111111111111111111111111111111111111112', 1);
      console.log('[provider] Helius credits restored — resuming monitor loop');
      monitorLoop.resume();
    } catch (err) {
      if (err instanceof HeliusCreditExhaustedError) {
        // Still exhausted — double delay (cap at 60m) and try again
        delayMs = Math.min(delayMs * 2, CREDIT_EXHAUSTION_MAX_DELAY_MS);
        probe();
      } else {
        // Different error (network, auth) — don't resume, but don't keep probing
        console.error('[provider] Helius probe failed with unexpected error:', err instanceof Error ? err.message : err);
      }
    }
  };

  probe();
}

export function createProviderRouter(): ProviderRouter {
  const heliusFetcher = createHeliusFetcher();

  // Wrap HeliusProvider to intercept credit exhaustion before ProviderRouter handles it
  const heliusProvider = new HeliusProvider(heliusFetcher);
  const heliusProviderWrapped: RpcProvider = {
    fetchSwapHistory: async (address, afterTimestamp) => {
      try {
        return await heliusProvider.fetchSwapHistory(address, afterTimestamp);
      } catch (err) {
        if (err instanceof HeliusCreditExhaustedError) {
          console.warn('[provider] Helius credit exhaustion detected — pausing monitor loop');
          monitorLoop.pause();
          startCreditExhaustionProbe(heliusFetcher);
        }
        throw err;  // Re-throw so ProviderRouter falls back to Shyft for this cycle
      }
    },
    fetchEarlySwapsForMint: async (mint, limit, sortOrder) => {
      try {
        return await heliusProvider.fetchEarlySwapsForMint(mint, limit, sortOrder);
      } catch (err) {
        if (err instanceof HeliusCreditExhaustedError) {
          console.warn('[provider] Helius credit exhaustion detected — pausing monitor loop');
          monitorLoop.pause();
          startCreditExhaustionProbe(heliusFetcher);
        }
        throw err;
      }
    },
    fetchOnePage: async (address, limit) => {
      try {
        return await heliusProvider.fetchOnePage(address, limit);
      } catch (err) {
        if (err instanceof HeliusCreditExhaustedError) {
          console.warn('[provider] Helius credit exhaustion detected — pausing monitor loop');
          monitorLoop.pause();
          startCreditExhaustionProbe(heliusFetcher);
        }
        throw err;
      }
    },
  };

  const providers: RpcProvider[] = [heliusProviderWrapped];
  // ... rest of createProviderRouter unchanged
}
```
