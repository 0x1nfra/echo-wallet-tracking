    Add `getStatus()` public method:
    ```ts
    getStatus(): Array<{ index: number; name: string; state: 'active' | 'cooldown'; lastError: string | null }> {
      return this.providers.map((provider, i) => ({
        index: i,
        name: provider.constructor.name,
        state: this.isOnCooldown(i) ? 'cooldown' : 'active',
        lastError: this.lastError.get(i) ?? null,
      }));
    }
    ```

    **ProviderRouter factory (`src/fetchers/providers/index.ts`):**

    The `createProviderRouter()` function returns a `ProviderRouter` but the return type may be typed as `RpcProvider`. Update the return type to `ProviderRouter` so callers can access `getStatus()`. Check the existing return type annotation and update accordingly. Import `ProviderRouter` type if not already imported.

    Also add a module-level shared provider status store used by the /admin route and /status Telegram command:

    ```ts
    // Module-level store for last-known provider status (consumed by /admin and /status)
    let _lastProviderStatus: Array<{ name: string; state: string; lastError: string | null }> = [];

    export function getSharedProviderStatus(): Array<{ name: string; state: string; lastError: string | null }> {
      return _lastProviderStatus;
    }

    export function updateSharedProviderStatus(router: ProviderRouter): void {
      _lastProviderStatus = router.getStatus();
    }
    ```

    Place the two exported functions after the `createProviderRouter()` function definition.
