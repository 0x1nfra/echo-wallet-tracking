  <action>
    Create `src/api/routes/admin.ts` as a Fastify plugin following the pattern of existing routes (e.g. `src/api/routes/status.ts`):

    ```ts
    import type { FastifyInstance } from 'fastify';
    import { db } from '../../db/index.js';
    import { sourcing_log } from '../../db/schema.js';
    import { desc } from 'drizzle-orm';

    export default async function adminRoutes(app: FastifyInstance): Promise<void> {
      app.get('/admin', async (_req, reply) => {
        // Use dynamic imports to avoid circular dependencies
        const { monitorLoop } = await import('../../commands/wallet.js');
        const { autoSourcer } = await import('../../monitor/index.js');

        // Get providerRouter status — it's not directly exported, use a module-level singleton
        // The createProviderRouter() is called per-cycle in loop.ts.
        // For /admin, create a lightweight status from the last known state.
        // Since ProviderRouter is instantiated per-cycle, we can't access its live instance.
        // Solution: export a shared providerRouter singleton from fetchers/providers/index.ts
        // OR simply show "Provider status: See Railway logs" as a fallback.
        // Best approach for Phase 15: expose provider status via a simple module-level store
        // that getStatus() updates. See action notes below.

        const monitorStats = {
          cycleCount: monitorLoop.cycleCount,
          lastCycleDurationMs: monitorLoop.lastCycleDurationMs,
          lastCycleCompletedAt: monitorLoop.lastCycleCompletedAt,
          // Stall detection: stalled if lastCycleCompletedAt > 5 minutes ago
          stalled: monitorLoop.lastCycleCompletedAt !== null
            && (Date.now() - monitorLoop.lastCycleCompletedAt) > 5 * 60 * 1000,
        };

        const sourcerStats = autoSourcer.getStats();

        // Recent sourcing log — last 10 entries
        const recentSourcingLog = db.select()
          .from(sourcing_log)
          .orderBy(desc(sourcing_log.polled_at))
          .limit(10)
          .all();

        // Provider status: use a shared singleton approach
        // Import the shared provider router from a new module-level export
        let providerStatus: Array<{ name: string; state: string; lastError: string | null }> = [];
        try {
          const { getSharedProviderStatus } = await import('../../fetchers/providers/index.js');
          providerStatus = getSharedProviderStatus();
        } catch {
          providerStatus = [{ name: 'unavailable', state: 'unknown', lastError: null }];
        }

        return reply.view('admin.ejs', {
          monitorStats,
          sourcerStats,
          recentSourcingLog,
          providerStatus,
          now: Date.now(),
        });
      });
    }
    ```

    **Provider status:**

    `getSharedProviderStatus()` and `updateSharedProviderStatus()` are implemented in Plan 01 Task 2 (`src/fetchers/providers/index.ts`). This plan calls `getSharedProviderStatus()` via dynamic import in the /admin route handler — no additional implementation needed here.

