    Create `src/monitor/removal.ts` implementing the three removal policies and a `removeWallet` helper.

    **Imports needed:**
    - `eq` from drizzle-orm
    - `db` from `../db/index.js`
    - `wallets`, `removal_log` from `../db/schema.js`

    **Constants:**
    ```ts
    export const LOW_SCORE_THRESHOLD = 30;      // score below this counts toward streak
    export const LOW_SCORE_STREAK_LIMIT = 10;   // consecutive cycles before removal
    export const INACTIVITY_DAYS = 30;          // configurable inactivity window
    ```

    **`removeWallet(address: string, reason: string, label: string | null, scoreAtRemoval: number | null, detectionDetails: string | null): void`**
    - Sets `wallets.status = 'removed'` for the given address
    - Inserts into `removal_log`: `{ wallet_address: address, reason, label, score_at_removal: scoreAtRemoval, detection_details: detectionDetails, removed_by: 'auto' }`
    - Logs to stdout: `[monitor] auto-removed ${address}${label ? ` (${label})` : ''} — ${reason}`

    **`checkRemovalPolicies(walletAddress: string): boolean`** — returns `true` if wallet was removed
    This function is called after each wallet's fetch→detect→score pipeline completes.

    Read the wallet row: `db.select().from(wallets).where(eq(wallets.address, walletAddress)).get()`
    If wallet not found, return false.

    **Policy 1 — Confirmed scam (RMVL-02):**
    If `wallet.detection_status === 'confirmed_suspicious'`:
    - Call `removeWallet(wallet.address, 'confirmed_suspicious detection', wallet.label, wallet.score, wallet.detection_status)`
    - Return true

    **Policy 2 — Score streak (RMVL-01):**
    If `wallet.score !== null && wallet.score < LOW_SCORE_THRESHOLD`:
    - Increment `wallets.low_score_streak` by 1: `db.update(wallets).set({ low_score_streak: (wallet.low_score_streak ?? 0) + 1 }).where(...).run()`
    - Re-read the updated row
    - If `low_score_streak >= LOW_SCORE_STREAK_LIMIT`:
      - Call `removeWallet(wallet.address, \`score below ${LOW_SCORE_THRESHOLD} for ${LOW_SCORE_STREAK_LIMIT} consecutive cycles\`, wallet.label, wallet.score, null)`
      - Return true
    Else if `wallet.score !== null && wallet.score >= LOW_SCORE_THRESHOLD`:
    - Reset streak: `db.update(wallets).set({ low_score_streak: 0 }).where(...).run()`

    **Policy 3 — Inactivity (RMVL-03):**
    Check `wallet.last_trade_at`:
    - If `wallet.last_trade_at === null`: **skip this check entirely** — treat as "not yet tracked by the monitoring loop, not inactive". This null-guard is a safety net for wallets that have no swap data at all (backfill in migration 0004 covers existing data, but newly added wallets with zero trades must not be immediately evicted).
    - If `last_trade_at` is set AND older than `INACTIVITY_DAYS * 24 * 60 * 60 * 1000` ms ago:
      - Call `removeWallet(wallet.address, \`no trades for ${INACTIVITY_DAYS} days\`, wallet.label, wallet.score, null)`
      - Return true

    Return false (no removal).

    Export: `checkRemovalPolicies`, `removeWallet`, `LOW_SCORE_THRESHOLD`, `LOW_SCORE_STREAK_LIMIT`, `INACTIVITY_DAYS`

    **Important:** Fetch errors (when wallet pipeline throws) do NOT call `checkRemovalPolicies` — errors are caught and skipped in the loop layer, so streak counter is never incremented on fetch failure. This is enforced by calling `checkRemovalPolicies` only after a successful pipeline run.

    ```ts
    const CYCLE_INTERVAL_MS = 30_000;
    const STARTUP_STAGGER_MS = 200; // stagger wallet fetches on first cycle to avoid burst
    const CRASH_RESTART_DELAY_MS = 5_000;
    ```

    **`MonitorLoop` class:**

    Private state:
    - `paused: boolean = false`
    - `stopped: boolean = false`
    - `timer: ReturnType<typeof setTimeout> | null = null`
    - `cycleRunning: boolean = false`

    **`start(): void`**
    Sets `stopped = false`, `paused = false`.
    Logs: `[monitor] starting — cycle interval ${CYCLE_INTERVAL_MS / 1000}s`
    Calls `this.scheduleNextCycle(0)` to run first cycle immediately (delay=0).

    **`pause(): void`**
    Sets `paused = true`.
    Logs: `[monitor] paused — current cycle will drain`

    **`resume(): void`**
    Sets `paused = false`.
    Logs: `[monitor] resumed`
    Calls `this.scheduleNextCycle(CYCLE_INTERVAL_MS)`.

    **`stop(): void`**
    Sets `stopped = true`, `paused = false`.
    If `timer` is set, calls `clearTimeout(timer)`, sets `timer = null`.
    Logs: `[monitor] stopped`

    **`private scheduleNextCycle(delayMs: number): void`**
    If `stopped`, return.
    Sets `this.timer = setTimeout(() => this.tick(), delayMs)`

    **`private async tick(): Promise<void>`**
    ```ts
    this.timer = null;
    if (this.stopped || this.paused) return;
    try {
      await this.runCycle();
    } catch (err) {
      console.error('[monitor] cycle crashed:', err);
      // Auto-restart after delay
      setTimeout(() => {
        if (!this.stopped && !this.paused) this.scheduleNextCycle(CRASH_RESTART_DELAY_MS);
      }, CRASH_RESTART_DELAY_MS);
      return;
    }
    if (!this.stopped && !this.paused) {
      this.scheduleNextCycle(CYCLE_INTERVAL_MS);
    }
    ```

    **`async runCycle(): Promise<void>`**
    ```ts
    const startMs = Date.now();
    // Get all tracked wallets (status='tracked', history_complete=true)
    const trackedWallets = db.select({
      address: wallets.address,
      last_checked_at: wallets.last_checked_at,
    }).from(wallets)
      .where(inArray(wallets.status, ['tracked']))
      .all()
      .filter(w => w !== null);

    console.log(`[monitor] cycle start — ${trackedWallets.length} wallets`);

    const fetcher = createHeliusFetcher();
    let processed = 0;
    let removed = 0;
    let failed = 0;

    for (let i = 0; i < trackedWallets.length; i++) {
      const wallet = trackedWallets[i];
      if (this.stopped || this.paused) break;

      // Stagger fetches on startup to avoid burst (200ms between wallets)
      if (i > 0) {
        await new Promise(resolve => setTimeout(resolve, STARTUP_STAGGER_MS));
      }

      try {
        // Incremental fetch: only since last_checked_at (in seconds for Helius API)
        const afterTimestamp = wallet.last_checked_at
          ? Math.floor(wallet.last_checked_at / 1000)
          : 0;

        const rawTxs = await fetcher.fetchSwapHistory(wallet.address, afterTimestamp);

        if (rawTxs.length > 0) {
          // Parse and insert new swaps
          const parsedSwaps: any[] = [];
          for (const tx of rawTxs) {
            try {
              parsedSwaps.push(...parseSwaps([tx], wallet.address));
            } catch {
              // Parse errors are silently skipped in steady state
            }
          }
          const enrichedSwaps = applyFifo(parsedSwaps);

          db.transaction((tx) => {
            for (const swap of enrichedSwaps) {
              try {
                tx.insert(swaps).values(swap).run();
              } catch (err) {
                if (err instanceof Error && err.message.includes('UNIQUE constraint failed')) continue;
                throw err;
              }
            }
          });

          // Update last_trade_at: find most recent buy/sell timestamp in new swaps
          if (enrichedSwaps.length > 0) {
            const latestSwapTs = Math.max(...enrichedSwaps.map((s: any) => s.timestamp));
            db.update(wallets)
              .set({ last_trade_at: latestSwapTs * 1000 }) // convert seconds to ms
              .where(eq(wallets.address, wallet.address))
              .run();
          }
        }

        // Update last_checked_at to now (milliseconds)
        db.update(wallets)
          .set({ last_checked_at: Date.now() })
          .where(eq(wallets.address, wallet.address))
          .run();

        // Run detection and scoring conditionally
        await runDetectionIfNeeded(wallet.address);
        scoreWalletIfNeeded(wallet.address);

        // Check removal policies
        const wasRemoved = checkRemovalPolicies(wallet.address);
        if (wasRemoved) removed++;
        else processed++;

      } catch (err) {
        // Single-wallet failure: log and skip — do NOT increment low_score_streak
        console.error(`[monitor] failed to process ${wallet.address}:`, err instanceof Error ? err.message : err);
        failed++;
      }
    }

    const durationMs = Date.now() - startMs;
    console.log(`[monitor] cycle complete — ${processed} processed, ${removed} removed, ${failed} failed in ${durationMs}ms`);
    ```
