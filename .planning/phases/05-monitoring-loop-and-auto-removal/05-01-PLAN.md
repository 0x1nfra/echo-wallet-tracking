    ```sql
    ALTER TABLE wallets ADD COLUMN last_trade_at INTEGER;
    ALTER TABLE removal_log ADD COLUMN label TEXT;
    ALTER TABLE removal_log ADD COLUMN score_at_removal REAL;

    -- Backfill last_trade_at from existing swap data so inactivity removal

    -- Uses MAX(timestamp * 1000) to convert Helius seconds → milliseconds.

    SET last_trade_at = (
      SELECT MAX(s.timestamp * 1000)
      FROM swaps s
      WHERE s.wallet_address = wallets.address
    )
    WHERE EXISTS (
      SELECT 1 FROM swaps s WHERE s.wallet_address = wallets.address
    );
    ```

    ```ts
        // Exponential backoff on rate limit: 2s, 4s, 8s, 16s, 32s
        if (status === 429) {
          const delayMs = Math.pow(2, error.attemptNumber) * 1000;
          await new Promise(resolve => setTimeout(resolve, delayMs));
        }
      },
    }
    ```
